 30:     public function __construct($connection_query, $ssl_key = null, $ssl_certificate = null, $ssl_ca = null)
 31:     {
 32:         if (!class_exists("MongoDB\\Driver\\Manager")) {
 33:             throw new \Gishiki\ActiveRecord\DatabaseException("No MongoDB driver available: install the mongodb pecl extension", 5);
 34:         }
 35:         
 36:         //get the database name
 37:         $dbpath = explode('/', $connection_query, 2);
 38:         if (count($dbpath) > 1) {
 39:             $dbpath = $dbpath[1];
 40:         }
 41:         $lim = strpos($dbpath, '?');
 42:         if ($lim !== false) {
 43:             $dbpath = substr($dbpath, 0, $lim + 1);
 44:         }
 45:         $this->dbname = $dbpath;
 46:         
 47:         try {
 48:             $this->connection = new \MongoDB\Driver\Manager("mongodb://".$connection_query);
 49:         } catch (\MongoDB\Driver\Exception\InvalidArgumentException $ex) {
 50:             throw new \Gishiki\ActiveRecord\DatabaseException("Unable to open a valid MongoDB connection (".$ex->getCode().")", 2);
 51:         } catch (\MongoDB\Driver\Exception\RuntimeException $ex) {
 52:             throw new \Gishiki\ActiveRecord\DatabaseException("Unable to open a valid MongoDB connection (".$ex->getCode().") invalid connection string", 3);
 53:         }

 58:         //get the fully qualified namespace (e.g. "databaseName.collectionName")

131:     private static function where_compile(\Gishiki\ActiveRecord\RecordsSelector $where, &$filter, &$options)
132:     {
133:         $filter = array();
134:         $options = array();
135:         
136:         //get the list of selectors
137:         $selectors_property = new \ReflectionProperty($where, "selectors");
138:         $selectors_property->setAccessible(true);
139:         $fields_selectors = $selectors_property->getValue($where);
140:         
141:         //place selectors
142:         $i = 0;
143:         $fields = array_keys($fields_selectors);
144:         foreach ($fields as $field) {
145:             $relationship = \Gishiki\Algorithms\Manipulation::get_string_between($field, '[', ']');
146:             $fieldname = str_replace('['.$relationship.']', "", $field);
147:             $fieldvalue = $fields_selectors[$field];
148:             
149:             //HERE you should change relationship if the rdbms you are using doesn't support:
150:             // = != < <= >= > 
151: 
152:             $filter[$fieldname] = array();
153:             if ($relationship == "=") {
154:                 $filter[$fieldname]['$eq'] = $fieldvalue;
155:             } elseif ($relationship == ">=") {
156:                 $filter[$fieldname]['$gte'] = $fieldvalue;
157:             } elseif ($relationship == "<=") {
158:                 $filter[$fieldname]['$lte'] = $fieldvalue;
159:             } elseif ($relationship == ">") {
160:                 $filter[$fieldname]['$gt'] = $fieldvalue;
161:             } elseif ($relationship == "<") {
162:                 $filter[$fieldname]['$lt'] = $fieldvalue;
163:             } elseif ($relationship == "!=") {
164:                 $filter[$fieldname]['$ne'] = $fieldvalue;
165:             }
166:             
167:             $i++;
168:         }
169:         
170:         //get the limit
171:         $limit_property = new \ReflectionProperty($where, "limit");
172:         $limit_property->setAccessible(true);
173:         
174:         $limit = $limit_property->getValue($where);
175:         if ($limit != 0) {
176:             $options['limit'] = $limit;
177:         }
178:         
179:         //get the offset
180:         $offset_property = new \ReflectionProperty($where, "offset");
181:         $offset_property->setAccessible(true);
182:         
183:         $offset = $offset_property->getValue($where);
184:         if ($offset != 0) {
185:             $options['skip'] = $offset;
186:         }
187:     }

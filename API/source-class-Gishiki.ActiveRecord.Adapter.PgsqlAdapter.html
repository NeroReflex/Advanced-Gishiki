137:     private function where_compile(\Gishiki\ActiveRecord\RecordsSelector $where, &$sobstitution_table)
138:     {
139:         //setup the sobtitution table
140:         $sobstitution_table = array();

142:         //start compiling
143:         $where_clauses = " WHERE ";
144:         
145:         //get the list of selectors
146:         $selectors_property = new \ReflectionProperty($where, "selectors");
147:         $selectors_property->setAccessible(true);
148:         $fields_selectors = $selectors_property->getValue($where);
149:         
150:         //place selectors
151:         $i = 0;
152:         $fields = array_keys($fields_selectors);
153:         foreach ($fields as $field) {
154:             $relationship = \Gishiki\Algorithms\Manipulation::get_string_between($field, '[', ']');
155:             $fieldname = str_replace('['.$relationship.']', "", $field);
156:             
157:             //HERE you should change relationship if the rdbms you are using doesn't support:
158:             // = != < <= >= > 
159: 
160:             $where_clauses .= $fieldname." ".$relationship." ?";
161:             
162:             //place comme if needed
163:             if ($i < count($fields) - 2) {
164:                 $where_clauses .= ' AND ';
165:             }
166:             $sobstitution_table[] = $fields_selectors[$field];
167:             
168:             $i++;
169:         }
170:         
171:         //get the limit
172:         $limit_property = new \ReflectionProperty($where, "limit");
173:         $limit_property->setAccessible(true);
174:         
175:         $limit = $limit_property->getValue($where);
176:         if ($limit) {
177:             $sobstitution_table[] = $limit;

186:         if ($offset) {

 20: /**
 21:  * An helper class for string manipulation.
 22:  *
 23:  * Benato Denis <benato.denis96@gmail.com>
 24:  */
 25: abstract class Base64
 26: {
 27:     /**
 28:      * Create the Base64 binary-safe representation of the given message.
 29:      *
 30:      * The given message can be a binary unsafe string.
 31:      *
 32:      * Example of usage:
 33:      * <code>
 34:      * //this is the binary unsafe message
 35:      * $message = " ... ";
 36:      *
 37:      * //print the result
 38:      * var_dump(Base64::encode($message));
 39:      * </code>
 40:      *
 41:      * @param string $message the binary-unsafe message
 42:      * @param bool   $urlSafe the generated result doesn't contains special characters
 43:      *
 44:      * @return string the binary-safe representation of the given message
 45:      *
 46:      * @throws \InvalidArgumentException the given message is not represented as a string
 47:      */
 48:     public static function encode($message, $urlSafe = true)
 49:     {
 50:         //check for the message type
 51:         if (!is_string($message)) {
 52:             throw new \InvalidArgumentException('the binary usafe content must be given as a string');
 53:         }
 54: 
 55:         //get the base64 url unsafe
 56:         $encoded = base64_encode($message);
 57: 
 58:         //return the url safe version if requested
 59:         return ($urlSafe) ? rtrim(strtr($encoded, '+/=', '-_~'), '~') : $encoded;
 60:     }
 61: 
 62:     /**
 63:      * Get the binary-unsafe representation of the given base64-encoded message.
 64:      *
 65:      * This function is compatible with the php standard base64_encode and the
 66:      * framework Base64::encode( ... ).
 67:      *
 68:      * Example of usage:
 69:      * <code>
 70:      * //this is the binary unsafe message
 71:      * $message = " ... ";
 72:      *
 73:      * //print the input string (binary unsafe)
 74:      * var_dump(Base64::decode(Base64::encode($message)));
 75:      * </code>
 76:      *
 77:      * @param string $message a message base64 encoded
 78:      *
 79:      * @return string the message in a binary-unsafe format
 80:      *
 81:      * @throws \InvalidArgumentException the given message is not represented as a string
 82:      */
 83:     public static function decode($message)
 84:     {
 85:         //check for the message type
 86:         if (!is_string($message)) {
 87:             throw new \InvalidArgumentException('the base64 of a string is represented as another string');
 88:         }
 89: 
 90:         //is the base64 encoded in an URL safe format?
 91:         $url_safe = (strlen($message) % 4) || (strpos($message, '_') !== false) || (strpos($message, '~') !== false);
 92: 
 93:         //get the base64 encoded valid string and return the decode result
 94:         $validBase64 = ($url_safe) ?
 95:                 str_pad(strtr($message, '-_~', '+/='), strlen($message) + 4 - (strlen($message) % 4), '=', STR_PAD_RIGHT)
 96:                 : $message;
 97: 
 98:         return base64_decode($validBase64);
 99:     }
100: }

 50:     /**
 51:      * Serialize the current data collection.
 52:      *
 53:      * @param int $format an integer representing one of the allowed formats
 54:      * @throw  SerializationException         the error occurred while serializing the collection in json format
 55:      *
 56:      * @return string the collection serialized
 57:      */
 58:     public function serialize($format = self::JSON)
 59:     {
 60:         $result = '';
 61:         switch ($format) {
 62: 
 63:             case self::JSON:
 64:                 //try json encoding
 65:                 $result = json_encode($this->all(), JSON_PRETTY_PRINT);
 66: 
 67:                 //and check for the result
 68:                 if (json_last_error() != JSON_ERROR_NONE) {
 69:                     throw new SerializationException('The given data cannot be serialized in JSON content', 2);
 70:                 }
 71:                 break;
 72: 
 73:             case self::XML:
 74:                 $xml = new \Gishiki\Algorithms\XmlDomConstructor('1.0', 'utf-8');
 75:                 $xml->xmlStandalone = true;
 76:                 $xml->formatOutput = true;
 77:                 $xml->fromMixed($this->all());
 78:                 $xml->normalizeDocument();
 79:                 $result = str_replace('standalone="yes"?>', 'standalone="yes"?><data>', $xml->saveXML());
 80:                 $result .= "\n</data>";
 81:                 break;
 82: 
 83:             case self::YAML:
 84:                 $result = \Symfony\Component\Yaml\Yaml::dump($this->all());
 85:                 break;
 86: 
 87:             default:
 88:                 throw new SerializationException('Invalid serialization format selected', 7);
 89:         }

 91:         return $result;

 94:     /**
 95:      * Deserialize the given data collectionand create a serializable data collection.
 96:      *
 97:      * @param string|array|CollectionInterface $message the string containing the serialized data or the array of data
 98:      * @param int                              $format  an integer representing one of the allowed formats
 99:      *
100:      * @return SerializableCollection the deserialization result
101:      *
102:      * @throws DeserializationException the error preventing the data deserialization
103:      */
104:     public static function deserialize($message, $format = self::JSON)
105:     {
106:         if ((is_array($message)) || ($message instanceof CollectionInterface)) {
107:             return new self($message);
108:         } elseif ($format === self::JSON) {
109:             if (!is_string($message)) {
110:                 throw new DeserializationException('The given content is not a valid JSON content', 3);
111:             }
112: 
113:             //try decoding the string
114:             $nativeSerialization = json_decode($message, true, 512);
115: 
116:             //and check for the result
117:             if (json_last_error() != JSON_ERROR_NONE) {
118:                 throw new DeserializationException('The given string is not a valid JSON content', 1);
119:             }
120: 
121:             //the deserialization result MUST be an array
122:             $serializationResult = (is_array($nativeSerialization)) ? $nativeSerialization : [];
123: 
124:             //return the deserialization result if everything went right
125:             return new self($serializationResult);
126:         } elseif ($format === self::XML) {
127:             if (!is_string($message)) {
128:                 throw new DeserializationException('The given content is not a valid XML content', 3);
129:             }
130: 
131:             //resolve CDATA
132:             $messageParsed = preg_replace_callback('/<!\[CDATA\[(.*)\]\]>/', function ($matches) {
133:                 return trim(htmlspecialchars($matches[1]));
134:             }, $message);
135: 
136:             //load the xml from the cleaned string
137:             libxml_use_internal_errors(true);
138:             $xml = simplexml_load_string($messageParsed);
139: 
140:             if ((count(libxml_get_errors()) > 0)/* || (!$xml)*/) {
141:                 //clear the error list to avoid interferences
142:                 libxml_clear_errors();
143: 
144:                 throw new DeserializationException('The given content is not a valid XML content', 3);
145:             }
146: 
147:             //use the json engine to deserialize the object
148:             $string = json_encode($xml);
149:             $nativeSerialization = json_decode($string, true);
150: 
151:             //return the deserialization result
152:             return new self($nativeSerialization);
153:         } elseif ($format === self::YAML) {
154:             if (!is_string($message)) {
155:                 throw new DeserializationException('The given content is not a valid YAML content', 8);
156:             }
157: 
158:             //attempt to deserialize the yaml file
159:             $nativeSerialization = null;
160:             try {
161:                 $nativeSerialization = \Symfony\Component\Yaml\Yaml::parse($message, true, true);
162:             } catch (\Symfony\Component\Yaml\Exception\ParseException $ex) {
163:                 throw new DeserializationException('The given YAML content cannot be deserialized', 9);
164:             }
165: 
166:             //check for the result type
167:             if (!is_array($nativeSerialization)) {
168:                 throw new DeserializationException('The YAML deserialization result cannot be used to build a collection', 10);
169:             }
170: 
171:             //return the deserialization result
172:             return new self($nativeSerialization);
173:         }
174: 
175:         //impossible to serialize the message
176:         throw new DeserializationException('It is impossible to deserialize the given message', 2);
177:     }

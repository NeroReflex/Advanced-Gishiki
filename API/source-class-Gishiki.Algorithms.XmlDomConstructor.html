14: class XmlDomConstructor extends \DOMDocument
15: {
16:     /**
17:      * Constructs elements and texts from an array or string.
18:      * The array can contain an element's name in the index part
19:      * and an element's text in the value part.
20:      *
21:      * It can also creates an xml with the same element tagName on the same
22:      * level.
23:      *
24:      * ex:
25:      \verbatim
26:      <nodes>
27:      <node>text</node>
28:      <node>
29:      <field>hello</field>
30:      <field>world</field>
31:      </node>
32:      </nodes>
33:      \verbatim
34:      *
35:      *
36:      * Array should then look like:
37:      \verbatim
38:      array(
39:      "nodes" => array(
40:      "node" => array(
41:      0 => "text",
42:      1 => array(
43:      "field" => array (
44:      0 => "hello",
45:      1 => "world",
46:      ),
47:      ),
48:      ),
49:      ),
50:      );
51:      \endverbatim
52:      *
53:      * @param mixed      $mixed      An array or string.
54:      * @param DOMElement $domElement the element from where the array will be construct to.
55:      */
56:     public function fromMixed($mixed, \DOMElement $domElement = null)
57:     {
58:         $domElement = is_null($domElement) ? $this : $domElement;
59: 
60:         if (is_array($mixed)) {
61:             foreach ($mixed as $index => $mixedElement) {
62:                 if (is_int($index)) {
63:                     if ($index == 0) {
64:                         $node = $domElement;
65:                     } else {
66:                         $node = $this->createElement($domElement->tagName);
67:                         $domElement->parentNode->appendChild($node);
68:                     }
69:                 } else {
70:                     $node = $this->createElement($index);
71:                     $domElement->appendChild($node);
72:                 }
73: 
74:                 $this->fromMixed($mixedElement, $node);
75:             }
76:         } else {
77:             $domElement->appendChild($this->createTextNode($mixed));
78:         }
79:     }
80: }

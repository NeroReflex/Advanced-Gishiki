 29:          * Parse the cache server connection string
 30:          *
 31:          * @param $connectionString string the cache server connection string
 32:          * @return array the parsing results
 33:          */
 34:         public static function Parse($connectionString)/* : array*/
 35:         {
 36:             //create some empty connection details
 37:             $conectionDetails = [
 38:                 "server_type" => "",
 39:             ];
 40:             
 41:             //if the connection string is not empty.....
 42:             if (strlen($connectionString) > 0) {
 43:                 //try fetching the caching server type, address and port
 44:                 $strings = explode("://", $connectionString, 2);
 45:                 if ($strings[0] == "memcached") {
 46:                     //update what is going to be returned
 47:                     $conectionDetails = self::ParseMemcached($strings[1]);
 48:                 } elseif (($strings[0] == "directory") || ($strings[0] == "filesystem")) {
 49:                     //update what is going to be returned
 50:                     $conectionDetails = self::ParseFilesystem($strings[1]);
 51:                 }
 52:             }
 53:             
 54:             //return the connection details in form of an array
 55:             return $conectionDetails;
 56:         }

 58:         /**
 59:          * Get the address and the port of the given memcached server connection string.
 60:          * The string is given in form of "address:port" without "memcached://" prepended
 61:          *
 62:          * @param  string $memcachedConnectionString the connection string
 63:          * @return array  the connection details
 64:          */
 65:         public static function ParseMemcached($memcachedConnectionString = "")/* : array*/
 66:         {
 67:             //split the server into address and port
 68:             $explosion = explode(":", $memcachedConnectionString, 2);
 69: 
 70:             //provide a default connection if no port or no address were furnished
 71:             if ((!isset($explosion[0])) || ($explosion[0] == "")) {
 72:                 $explosion[0] = "localhost";
 73:             }
 74:             if ((!isset($explosion[1])) || ($explosion[1] == "")) {
 75:                 $explosion[1] = "11211";
 76:             }
 77: 
 78:             //return the split result
 79:             return [
 80:                 "server_type" => "memcached",
 81:                 "server_address" => $explosion[0],
 82:                 "server_port" => intval($explosion[1]),
 83:             ];
 84:         }
 85: 
 86:         /**
 87:          * Get the directory to be used as the caching entry.
 88:          * The string is given in form of "directory/" without "filesystem://" prepended
 89:          *
 90:          * @param  string $directory the connection string
 91:          * @return array  the connection details
 92:          */
 93:         public static function ParseFilesystem($directory = "")/* : array*/
 94:         {
 95:             $directoryPath = "".$directory;
 96:             if ($directoryPath[strlen($directoryPath) - 1] != DS) {
 97:                 $directoryPath .= DS;
 98:             }
 99: 
100:             //return the split result
101:             return [
102:                 "server_type" => "filesystem",
103:                 "directory" => $directoryPath,
104:             ];
105:         }

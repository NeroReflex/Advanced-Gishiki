 61:      * {@inheritdoc}
 62:      */
 63:     public function connect($details)
 64:     {
 65:         //check for argument type
 66:         if ((!is_string($details)) || (strlen($details) <= 0)) {
 67:             throw new \InvalidArgumentException('The connection query must be given as a non-empty string');
 68:         }
 69: 
 70:         //check for the pdo driver
 71:         if (!in_array('sqlite', \PDO::getAvailableDrivers())) {
 72:             throw new DatabaseException('No SQLite PDO driver', 0);
 73:         }
 74: 
 75:         //open the connection
 76:         try {
 77:             $this->connection = new \PDO('sqlite:'.$details);
 78:             $this->connection->setAttribute(\PDO::ATTR_ERRMODE, \PDO::ERRMODE_EXCEPTION);
 79: 
 80:             //the connection is opened
 81:             $this->connected = true;
 82:         } catch (\PDOException $ex) {
 83:             throw new DatabaseException('Error while opening the database connection:'.$ex->getMessage(), 1);
 84:         }
 85:     }

103:      * {@inheritdoc}
104:      */
105:     public function create($collection, $data)
106:     {
107:         //check for invalid database name
108:         if ((!is_string($collection)) || (strlen($collection) <= 0)) {
109:             throw new \InvalidArgumentException('The name of the table must be given as a non-empty string');
110:         }
111: 
112:         //check for invalid data collection
113:         if ((!is_array($data)) && (!($data instanceof CollectionInterface))) {
114:             throw new \InvalidArgumentException('The data to be written on the database must be given as a collection');
115:         }
116: 
117:         //check for closed database connection
118:         if (!$this->connected()) {
119:             throw new DatabaseException('The database connection must be opened before executing any operation', 2);
120:         }
121: 
122:         //get an associative array of the input data
123:         $adaptedData = ($data instanceof GenericCollection) ? $data->all() : $data;
124: 
125:         //build the sql query
126:         $queryBuilder = new SQLQueryBuilder();
127:         $queryBuilder->insertInto($collection)->values($adaptedData);
128: 
129:         //open a new statement and execute it
130:         try {
131:             //prepare a statement with that safe sql string
132:             $stmt = $this->connection->prepare($queryBuilder->exportQuery());
133: 
134:             //execute the statement resolving placeholders
135:             $stmt->execute($queryBuilder->exportParams());
136: 
137:             //as per documentation return the id of the last inserted row
138:             return $this->connection->lastInsertId();
139:         } catch (\PDOException $ex) {
140:             throw new DatabaseException('Error while performing the creation operation: '.$ex->getMessage(), 3);
141:         }
142:     }

 24: /**
 25:  * This utility is useful to create sql queries for various RDBMS.
 26:  *
 27:  * An example of usage can be:
 28:  * <code>
 29:  * $queryBuilder = new SQLQueryBuilder();
 30:  * $queryBuilder->insertInto("users")->values([
 31:  *      "name" => "Mario",
 32:  *      "surname" => "Rossi",
 33:  *      "password" => Hashing\Algorithms::hash(" __= N0 r4inb0w t4bl3 ==__" . "Mario's password", Hashing\Algorithms::SHA512);
 34:  *      "sex" => 0,
 35:  *      "height" => 1.70
 36:  * ]);
 37:  *
 38:  * //INSERT INTO "users" (name, surname, password, sex, height) VALUES (?, ?, ?, ?, ?)
 39:  * $sql = $queryBuilder->exportQuery();
 40:  *
 41:  * // array( "Mario", "Rossi", ".....", 0, 1.70 )
 42:  * $params = $queryBuilder->exportParams();
 43:  * </code>
 44:  *
 45:  * @author Benato Denis <benato.denis96@gmail.com>
 46:  */

 59:     /**
 60:      * Beautify an SQL query.
 61:      *
 62:      * @param  string $query your ugly SQL query
 63:      * @return string        your beautiful SQL query
 64:      */
 65:     public static function beautify($query)
 66:     {
 67:         $count = 1;
 68:         while ($count > 0) {
 69:             $query = str_replace('  ', ' ', $query, $count);
 70:         }
 71: 
 72:         $count = 1;
 73:         while ($count > 0) {
 74:             $query = str_replace('( ', '(', $query, $count);
 75:         }
 76: 
 77:         $count = 1;
 78:         while ($count > 0) {
 79:             $query = str_replace(' )', ')', $query, $count);
 80:         }
 81: 
 82:         $count = 1;
 83:         while ($count > 0) {
 84:             $query = str_replace('?, ?', '?,?', $query, $count);
 85:         }
 86: 
 87:         return trim($query);
 88:     }

 90:     /**
 91:      * Append to the current SQL the given text.
 92:      *
 93:      * Placeholders are question marks: ? and the value must be registered using
 94:      * the appendToParams function.
 95:      *
 96:      * @param string $sql the SQL with '?' placeholders
 97:      */
 98:     protected function appendToQuery($sql)
 99:     {
100:         $this->sql .= $sql;
101:     }
102: 
103:     /**
104:      * This is a collection of raw values that the PDO will replace to ?
105:      * on the SQL query.
106:      *
107:      * @param mixed $newParams an array of values or the value to be replaced
108:      */
109:     protected function appendToParams($newParams)
110:     {
111:         //this is used to recreate an array that doesn't conflicts with the $this->params when merging them
112:         if (gettype($newParams) == 'array') {
113:             $temp = [];
114: 
115:             foreach ($newParams as $currentKey => $currentValue) {
116:                 $temp[$currentKey + count($this->params)] = $currentValue;
117:             }
118: 
119:             $newParams = $temp;
120:         }
121:         //the result will be something like:
122:         // [0] => x1, [1] => x2
123:         // [5] => x1, [6] => x2
124: 
125:         (is_array($newParams)) ? $this->params = array_merge($this->params, $newParams) : array_push($this->params, $newParams);
126:     }

211:     /**
212:      * Add WHERE col1 = ? OR col2 <= ? ....... to the SQL query.
213:      *
214:      * @param SelectionCriteria $where the selection criteria
215:      *
216:      * @return \Gishiki\Database\Adapters\Utils\SQLQueryBuilder the updated sql builder
217:      */
218:     public function &where(SelectionCriteria $where)
219:     {
220:         //execute the private function 'export'
221:         $exportMethod = new \ReflectionMethod($where, 'export');
222:         $exportMethod->setAccessible(true);
223:         $resultExported = $exportMethod->invoke($where);
224: 
225:         if (count($resultExported['historic']) > 0) {
226:             $this->appendToQuery('WHERE ');
227: 
228:             $first = true;
229:             foreach ($resultExported['historic'] as $current) {
230:                 $conjunction = '';
231: 
232:                 $arrayIndex = $current & (~SelectionCriteria::AND_Historic_Marker);
233:                 $arrayConjunction = '';
234: 
235:                 //default is an OR
236:                 $conjunction = (!$first) ? ' OR ' : ' ';
237:                 $arrayConjunction = 'or';
238: 
239:                 //change to AND where necessary
240:                 if (($current & (SelectionCriteria::AND_Historic_Marker)) != 0) {
241:                     $conjunction = (!$first) ? ' AND ' : ' ';
242:                     $arrayConjunction = 'and';
243:                 }
244: 
245:                 $fieldName = $resultExported['criteria'][$arrayConjunction][$arrayIndex][0];
246:                 $fieldRelation = $resultExported['criteria'][$arrayConjunction][$arrayIndex][1];
247:                 $fieldValue = $resultExported['criteria'][$arrayConjunction][$arrayIndex][2];
248: 
249:                 //assemble the query
250:                 $qmarks = '';
251:                 $parentOpen = '';
252:                 $parentClose = '';
253:                 if (is_array($fieldValue)) {
254:                     $qmarks = str_repeat(' ?,', count($fieldValue) - 1);
255:                     $parentOpen = '(';
256:                     $parentClose = ')';
257:                 }
258:                 $this->appendToQuery($conjunction.$fieldName.' '.$fieldRelation.' '.$parentOpen.$qmarks.' ?'.$parentClose.' ');
259:                 $this->appendToParams($fieldValue);
260: 
261:                 $first = false;
262:             }
263:         }
264: 
265:         //chain functions calls
266:         return $this;
267:     }

319:     public function &limitOffsetOrderBy(ResultModifier $mod)
320:     {
321:         //execute the private function 'export'
322:         $exportMethod = new \ReflectionMethod($mod, 'export');
323:         $exportMethod->setAccessible(true);
324:         $resultExported = $exportMethod->invoke($mod);
325: 
326:         //append limit if needed
327:         if ($resultExported['limit'] > 0) {
328:             $this->appendToQuery('LIMIT '.$resultExported['limit'].' ');

333:             $this->appendToQuery('OFFSET '.$resultExported['skip'].' ');
334:         }
335: 
336:         //append order if needed
337:         if (count($resultExported['order']) > 0) {
338:             $this->appendToQuery('ORDER BY ');

20: /**
21:  * Represent the database manager of the entire framework.
22:  *
23:  * @author Benato Denis <benato.denis96@gmail.com>
24:  */
25: abstract class DatabaseManager
26: {
27:     private static $connections = array();
28: 
29:     //used to give a sencond name to an adapter
30:     private static $adaptersMap = [
31:         // 'Firstname' => 'Secondname'
32:     ];
33: 
34:     /**
35:      * Create a new database connection and store the newly generated connection.
36:      *
37:      * @param string $connectionName   the name of the database connection
38:      * @param string $connectionString the connection string
39:      *
40:      * @throws \InvalidArgumentException invalid name or connection string
41:      *
42:      * @return DatabaseInterface the connected database instance
43:      */
44:     public static function connect($connectionName, $connectionString)
45:     {
46:         //check for malformed input
47:         if ((!is_string($connectionName)) || (strlen($connectionName) <= 0) || (!is_string($connectionString)) || (strlen($connectionString) <= 0)) {
48:             throw new \InvalidArgumentException('The connection name and the connection details must be given as two string');
49:         }
50: 
51:         //get the adapter name
52:         $temp = explode('://', $connectionString);
53:         $adapterTemp = ucfirst($temp[0]);
54:         $adapter = (array_key_exists($adapterTemp, self::$adaptersMap)) ?
55:             self::$adaptersMap[$adapter] : $adapterTemp;
56:         $connectionQuery = $temp[1];
57: 
58:         try {
59:             //reflect the adapter
60:             $reflectedAdapter = new \ReflectionClass('Gishiki\\Database\\Adapters\\'.$adapter);
61: 
62:             //and use the adapter to estabilish the database connection and return the connection handler
63:             self::$connections[$connectionName] = $reflectedAdapter->newInstance($connectionQuery);
64: 
65:             return self::$connections[$connectionName];
66:         } catch (\ReflectionException $ex) {
67:             throw new DatabaseException('The given connection query requires an nonexistent adapter', 0);
68:         }
69:     }
70: 
71:     /**
72:      * Retrieve the connection with the given name from the list of performed conenctions.
73:      * If the name is not specified the default one is retrieved.
74:      *
75:      * @param string $connectionName the name of the selected connection
76:      *
77:      * @return DatabaseInterface the connected database instance
78:      *
79:      * @throws \InvalidArgumentException the collection name has not be given as a string
80:      * @throws DatabaseException         the given connection name is not registered as a valid collection
81:      */
82:     public static function retrieve($connectionName = 'default')
83:     {
84:         //check for malformed input
85:         if ((!is_string($connectionName)) || (strlen($connectionName) <= 0)) {
86:             throw new \InvalidArgumentException('The name of the connection to be retrieved must be given as a string');
87:         }
88: 
89:         //check if the connection was enstabilished
90:         if (!array_key_exists($connectionName, self::$connections)) {
91:             throw new DatabaseException("The given connection doesn't exists", 1);
92:         }
93: 
94:         //return the enstabilished conenction
95:         return self::$connections[$connectionName];
96:     }
97: }

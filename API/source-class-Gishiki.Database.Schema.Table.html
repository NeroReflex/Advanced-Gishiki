 22: /**
 23:  * Represent a table inside a database.
 24:  *
 25:  * @author Benato Denis <benato.denis96@gmail.com>
 26:  */
 27: final class Table
 28: {
 29:     /**
 30:      * @var string the name of the table
 31:      */
 32:     protected $name;
 33: 
 34:     /**
 35:      * @var array a list of columns inside the current database
 36:      */
 37:     protected $columns;
 38: 
 39:     /**
 40:      * Initialize a table with the given name.
 41:      * This function internally calls setName(), and you should catch
 42:      * exceptions thrown by that function.
 43:      *
 44:      * @param string $name the name of the table
 45:      */
 46:     public function __construct($name)
 47:     {
 48:         $this->name = '';
 49:         $this->columns = [];
 50:         $this->foreignKeys = [];
 51:         $this->setName($name);
 52:     }
 53: 
 54:     /**
 55:      * Add a column to the current table.
 56:      *
 57:      * @param \Gishiki\Database\Schema\Column $col the column to be added
 58:      *
 59:      * @return \Gishiki\Database\Schema\Table a reference to the modified table
 60:      *
 61:      * @throws DatabaseException A table with the same name already exists
 62:      */
 63:     public function &addColumn(Column &$col)
 64:     {
 65:         foreach ($this->columns as $currentCol) {
 66:             if (strcmp($col->getName(), $currentCol->getName()) == 0) {
 67:                 throw new DatabaseException('A Table cannot contain two columns with the same name', 140);
 68:             }
 69:         }
 70: 
 71:         $this->columns[] = $col;
 72: 
 73:         return $this;
 74:     }
 75: 
 76:     /**
 77:      * Return the list of columns inside the current table.
 78:      *
 79:      * @return array thelist of columns
 80:      */
 81:     public function getColumns()
 82:     {
 83:         return $this->columns;
 84:     }
 85: 
 86:     /**
 87:      * Change the name of the current table.
 88:      *
 89:      * @param string $name the name of the table
 90:      *
 91:      * @return \Gishiki\Database\Schema\Table a reference to the modified table
 92:      *
 93:      * @throws \InvalidArgumentException the table name is invalid
 94:      */
 95:     public function &setName($name)
 96:     {
 97:         //avoid bad names
 98:         if ((!is_string($name)) || (strlen($name) <= 0)) {
 99:             throw new \InvalidArgumentException('The name of a table must be expressed as a non-empty string');
100:         }
101: 
102:         $this->name = $name;
103: 
104:         return $this;
105:     }
106: 
107:     /**
108:      * Retrieve the name of the table.
109:      *
110:      * @return string the table name
111:      */
112:     public function getName()
113:     {
114:         return $this->name;
115:     }

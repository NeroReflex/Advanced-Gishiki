 20:     /**
 21:      * This class provides the core Ahead-Of-Time component:
 22:      * the code generator.
 23:      * 
 24:      * This component is used to generate PHP classes out of the given
 25:      * database structure, binding activerecord functionalities to that
 26:      * generated classes.
 27:      *
 28:      * @author Benato Denis <benato.denis96@gmail.com>
 29:      */
 30:     class ModelBuilder {
 31:         //this is the database structure deducted by the analyzer
 32:         private $database_structure;
 33:         
 34:         //this is the list of tables
 35:         private $tables;
 36:         
 37:         //was the error check performed
 38:         private $checked;
 39:         
 40:         /**
 41:          * Create a code generator over the given database structure
 42:          * 
 43:          * @param \Gishiki\ORM\Common\Database $dbStructure the result of the analysis
 44:          */
 45:         public function __construct(\Gishiki\ORM\Common\Database &$dbStructure) {
 46:             //no error check performed (yet)
 47:             $this->checked = FALSE;
 48:                     
 49:             //this is the table 
 50:             $this->tables = array();
 51:             
 52:             if ($dbStructure != NULL) //store a reference to the database structure if it is valid
 53:             {   $this->database_structure = $dbStructure;   }
 54:         }
 55:         
 56:         /**
 57:          * Perform an accurate error check on the result of the analyzer and 
 58:          * prepare the structure for future steps
 59:          * 
 60:          * @throws \Gishiki\ORM\ModelBuilding\ModelBuildingException the caught error
 61:          */
 62:         public function ErrorsCheck() {
 63:             $table_names = array();
 64:             
 65:             //check for the name validity of the database
 66:             if (!$this->database_structure->hasValidName())
 67:             {   throw new \Gishiki\ORM\ModelBuilding\ModelBuildingException("'".$this->database_structure."' is not a valid name for a database", 10); }
 68:                 
 69:             //check if each table has a valid name
 70:             foreach ($this->database_structure as $table_index => $current_table) {
 71:                 if (!$current_table->hasValidName()) //check for the name validity
 72:                 {   throw new \Gishiki\ORM\ModelBuilding\ModelBuildingException("in database '".$this->database_structure."': ".$current_table."' is not a valid name for a database table", 11); }
 73:                 else {
 74:                     //check if a table has two field with the same name
 75:                     if (!in_array($current_table, $table_names))
 76:                     { //check if a table with the same name have been already added
 77:                         $table_names[] = $current_table;
 78:                         
 79:                         //check for tables without a primary key
 80:                         if (!$current_table->hasPrimaryKey())
 81:                         {   throw new \Gishiki\ORM\ModelBuilding\ModelBuildingException("in database '".$this->database_structure."': in table ".$current_table."': the table doesn't have a primary key", 12);    }
 82:                     }
 83:                     else
 84:                     {   throw new \Gishiki\ORM\ModelBuilding\ModelBuildingException("in database '".$this->database_structure."': the ".$current_table."' name is used for two or many tables", 13);   }
 85:                 }
 86:             }
 87:             
 88:             //errors check performed
 89:             $this->checked = TRUE;
 90:         }
 91:         
 92:         /**
 93:          * Perform the PHP code generation, doesn't prepend '<?php' to the code.
 94:          * 
 95:          * The code is not eval()'d, but can be directly eval()'d: this
 96:          * is usefull when a bad storage slows down the server Gishiki is 
 97:          * running from
 98:          * 
 99:          * @return string the generated PHP code
100:          */
101:         public function Compile() {
102:             //start up the code generation with a code opening tag
103:             $generated_code = /*"<?php\n".*/ "";
104:             
105:             //emplace a notice
106:             $generated_code .= "/*******************************************".PHP_EOL;
107:             $generated_code .= " automatically generated by the Gishiki ORM ".PHP_EOL;
108:             $generated_code .= "   do not edit unless extremely necessary   ".PHP_EOL;
109:             $generated_code .= "********************************************/".PHP_EOL.PHP_EOL.PHP_EOL;
110:             
111:             //generate a class from each table
112:             foreach ($this->database_structure as $current_table) {
113:                 $model_name = rtrim("".$current_table, "s");
114:                 
115:                 //add the base structure of the class
116:                 $generated_code .= "/**".PHP_EOL;
117:                 $generated_code .= " * The model built around the description of the table \"".$model_name."\" ".PHP_EOL;
118:                 $generated_code .= " * of the database named \"".$this->database_structure."\".".PHP_EOL;
119:                 $generated_code .= " *".PHP_EOL;
120:                 $generated_code .= " * @author nobody <auto-generated>".PHP_EOL;
121:                 $generated_code .= " */".PHP_EOL;
122:                 $generated_code .= "class ".$model_name." extends \ActiveRecord\Model {".PHP_EOL.PHP_EOL;
123:                 
124:                 //add the database connection
125:                 $generated_code .= "    /**".PHP_EOL;
126:                 $generated_code .= "     * get the name of the connection to the proper database".PHP_EOL;
127:                 $generated_code .= "     * ".PHP_EOL;
128:                 $generated_code .= "     * @var string the name of the connection to be used".PHP_EOL;
129:                 $generated_code .= "     */".PHP_EOL;
130:                 $generated_code .= "    static \$connection = '".$this->database_structure->getConnection()."';".PHP_EOL.PHP_EOL;
131:                 
132:                 //add the class destructor
133:                 $generated_code .= "    /**".PHP_EOL;
134:                 $generated_code .= "     * Release every used resource and save the model to the database".PHP_EOL;
135:                 $generated_code .= "     */".PHP_EOL;
136:                 $generated_code .= "    public function __destruct() {".PHP_EOL;
137:                 $generated_code .= "        //auto-update the database and shutdown operations".PHP_EOL;
138:                 $generated_code .= "        parent::__destruct();".PHP_EOL;
139:                 $generated_code .= "    }".PHP_EOL.PHP_EOL;
140:                 
141:                 //create getters and setters
142:                 foreach ($current_table as $current_field) {
143:                     //this is the field name in the database
144:                     $field_name = "".$current_field;
145:                     
146:                     //this is the field name for comments
147:                     $comment_field_name = str_replace("_", " ", $field_name);
148:                     
149:                     //this is the field name for function names
150:                     $field_upchar_name = ucfirst($field_name);
151:                     
152:                     //type subsystem
153:                     $type_name = "";
154:                     $type_filter = "";
155:                     $type_prepend = "";
156:                     switch ($current_field->getDataType()) {
157:                         case \Gishiki\ORM\Common\DataType::BOOLEAN:
158:                             $type_name = "boolean";
159:                             $type_filter = "boolval";
160:                             break;
161:                         case \Gishiki\ORM\Common\DataType::FLOAT:
162:                             $type_name = "float";
163:                             $type_filter = "floatval";
164:                             break;
165:                         case \Gishiki\ORM\Common\DataType::INTEGER:
166:                             $type_name = "integer";
167:                             $type_filter = "intval";
168:                             break;
169:                         case \Gishiki\ORM\Common\DataType::STRING:
170:                             $type_name = "string";
171:                             $type_filter = "strval";
172:                             break;
173:                         case \Gishiki\ORM\Common\DataType::DATETIME:
174:                             $type_name = "\ActiveRecord\DateTime";
175:                             $type_prepend = $type_name;
176:                             break;
177:                         default:
178:                             $type_name = "mixed";
179:                             $type_filter = "";
180:                             break;
181:                     } $type_prepend .= " ";
182:                     
183:                     //avoid breaking ActiveRecord getters and setters
184:                     if ($field_upchar_name[0] != '_') {
185:                         //generate the getter
186:                         $generated_code .= "    /**".PHP_EOL;
187:                         $generated_code .= "     * Get the ".$comment_field_name." of the current ".$model_name."".PHP_EOL;
188:                         $generated_code .= "     *".PHP_EOL;
189:                         $generated_code .= "     * @return ".$type_name." the ".$comment_field_name." of the ".$model_name.PHP_EOL;
190:                         $generated_code .= "     */".PHP_EOL;
191:                         $generated_code .= "    public function get".$field_upchar_name."() {".PHP_EOL;
192:                         $generated_code .= "        return ".$type_filter."(\$this->read_attribute('".$field_name."'));".PHP_EOL;
193:                         $generated_code .= "    }".PHP_EOL.PHP_EOL;
194: 
195:                         //generate the setter
196:                         $generated_code .= "    /**".PHP_EOL;
197:                         $generated_code .= "     * Set the ".$comment_field_name." of the current ".$model_name."".PHP_EOL;
198:                         $generated_code .= "     *".PHP_EOL;
199:                         $generated_code .= "     * @param mixed \$val the new ".$comment_field_name." of the ".$model_name.PHP_EOL;
200:                         $generated_code .= "     * @return ".$type_name." the new ".$comment_field_name." of the ".$model_name." casted to the correct type".PHP_EOL;
201:                         $generated_code .= "     */".PHP_EOL;
202:                         $generated_code .= "    public function set".$field_upchar_name."(".$type_prepend."\$val) {".PHP_EOL;
203:                         $generated_code .= "        return \$this->assign_attribute('".$field_name."', ".$type_filter."(\$val));".PHP_EOL;
204:                         $generated_code .= "    }".PHP_EOL.PHP_EOL.PHP_EOL;
205:                     }
206:                 }
207:                 
208:                 
209:                 $generated_code .= "}".PHP_EOL.PHP_EOL.PHP_EOL;
210:             }
211:             
212:             //return the generated php code
213:             return $generated_code;
214:         }

216:     }

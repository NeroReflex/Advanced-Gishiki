 23: /**
 24:  * Give runtime support to the pipeline component.
 25:  * 
 26:  * @author Benato Denis <benato.denis96@gmail.com>
 27:  */
 28: abstract class PipelineSupport
 29: {
 30:     /**
 31:      * @var \Gishiki\Database\DatabaseInterface the connection to the database
 32:      */
 33:     private static $connectionHandler = null;
 34: 
 35:     /**
 36:      * @var string the name of the table
 37:      */
 38:     private static $tableName = null;
 39: 
 40:     /**
 41:      * @var bool have the PipelineSupport being initialized?
 42:      */
 43:     private static $initialized = false;
 44: 
 45:     /**
 46:      * @var PipelineRuntime|null the currently active runtime or null 
 47:      */
 48:     private static $activeRuntime = null;
 49: 
 50:     /**
 51:      * Internal use ONLY!
 52:      * Setup the PipelineSupport from the global Environment.
 53:      * 
 54:      * @param string $connectionName the name of the database connection
 55:      * @param string $tableName      the name of the table inside the database connection
 56:      *
 57:      * @throws DatabaseException         the exception occurred while fetching the database connection
 58:      * @throws \InvalidArgumentException given arguments are not valid strings
 59:      */
 60:     public static function Initialize($connectionName, $tableName)
 61:     {
 62:         //only works for the first time (prevent user from doing bad things)
 63:         if (self::$initialized) {
 64:             return;
 65:         }
 66:         self::$initialized = true;
 67: 
 68:         if ((!is_string($connectionName)) || (!is_string($tableName))) {
 69:             throw new \InvalidArgumentException('Database connection name and table must be given as strings');
 70:         }
 71: 
 72:         //retrieve the database connection
 73:         self::$connectionHandler = \Gishiki\Database\DatabaseManager::Retrieve($connectionName);
 74: 
 75:         //store the table name
 76:         self::$tableName = $tableName;
 77:     }
 78: 
 79:     /**
 80:      * Internal use ONLY!
 81:      * Flag the given PipelineRuntime as the currently active one.
 82:      * 
 83:      * @param PipelineRuntime $runtime
 84:      */
 85:     public static function RegisterRuntime(PipelineRuntime &$runtime)
 86:     {
 87:         //get the runtime
 88:         self::$activeRuntime = /*&*/$runtime;
 89:     }
 90: 
 91:     /**
 92:      * Internal use ONLY!
 93:      * The currently active PipelineRuntime is marked as not-active-anymore.
 94:      */
 95:     public static function UnregisterRuntime()
 96:     {
 97:         //get the runtime
 98:         self::$activeRuntime = null;
 99:     }
100: 
101:     /**
102:      * Abort the current pipeline execution saving the reason.
103:      * 
104:      * @param string $message the message to be saved as the reason of the abort
105:      *
106:      * @throws \InvalidArgumentException the given message is not a string
107:      */
108:     public static function Abort($message = '')
109:     {
110:         if (!is_string($message)) {
111:             throw new \InvalidArgumentException('The abort message must be given as a string');
112:         }
113: 
114:         //throw the abort signal
115:         throw new PipelineAbortSignal($message);
116:     }
117: 
118:     /**
119:      * Save the currently active PipelineRuntime.
120:      * 
121:      * @throws PipelineException no pipeline is going to be saved
122:      */
123:     public static function saveCurrentPupeline()
124:     {
125:         if (is_null(self::$activeRuntime)) {
126:             throw new PipelineException('No pipeline currently flagged as active', 1);
127:         }
128: 
129:         //reflect the PipelineRuntime currently active
130:         $pipelineRuntimeReflected = new \ReflectionObject(self::$activeRuntime);
131: 
132:         //generate the data to be saved
133:         $data = [
134:             'uniqID' => self::$activeRuntime->getUniqueID(),
135:             'status' => self::$activeRuntime->getStatus(),
136:             'type' => self::$activeRuntime->getType(),
137:             'priority' => self::$activeRuntime->getPriority(),
138:             'creationTime' => self::$activeRuntime->getCreationTime(),
139:             'completionReports' => self::$activeRuntime->getExecutionReport()->all(),
140:             'pipeline' => self::$activeRuntime->getPipelineName(),
141:             'abortMessage' => self::$activeRuntime->getAbortMessage(),
142:             'serializableCollection' => self::$activeRuntime->getDataCollection()->all(),
143:         ];
144: 
145:         //identify the pipeline if already saved
146:         $selector = new \Gishiki\Database\SelectionCriteria();
147:         $selector->EqualThan('uniqID', self::$activeRuntime->getUniqueID());
148: 
149:         if (self::$connectionHandler->Fetch(self::$tableName, $selector)->count() > 0) {
150:             //save the pipeline status in an already existing record
151:              self::$connectionHandler->Update(self::$tableName, $data, $selector);
152: 
153:             return;
154:         }
155: 
156:         //save the pipeline status in a new record
157:         self::$connectionHandler->Insert(self::$tableName, $data);
158:     }
159:     
160:     /**
161:      * Forward the request to PipelineSupport.
162:      * 
163:      * @param string $uniqueID the unique ID of the PipelineRuntime
164:      * @return PipelineRuntime the restored runtime
165:      * @throws PipelineException the given unique ID is not valid
166:      * @throws \InvalidArgumentException the unique ID is not valid or the pipeline is executing
167:      */
168:     public static function Restore($uniqueID)
169:     {
170:         if ((!is_string($uniqueID)) || (strlen($uniqueID) <= 0)) {
171:             throw new \InvalidArgumentException("The unique pipeline ID is not a valid string");
172:         }
173:         
174:         //identify the pipeline if already saved
175:         $selector = new \Gishiki\Database\SelectionCriteria();
176:         $selector->EqualThan('uniqID', $uniqueID);
177: 
178:         //check for the unique ID
179:         $pipelineCollectionFetched = self::$connectionHandler->Fetch(self::$tableName, $selector);
180:         if ($pipelineCollectionFetched->count() != 1) {
181:             throw new \Gishiki\Pipeline\PipelineException("The given ID doesn't identify an unique pipeline runtime", 2);
182:         }
183:         
184:         //check se status of the pipeline
185:         $toRestore = $pipelineCollectionFetched->get(0)->GetData();
186:         if ($toRestore->get('status') == RuntimeStatus::WORKING) {
187:             throw new \Gishiki\Pipeline\PipelineException("The given ID matches a pipeline runtime that is executed elsewhere", 3);
188:         }
189:         
190:         //create an empty pipeline
191:         $pipelineClassReflection = new \ReflectionClass('Gishiki\\Pipeline\\PipelineRuntime');
192:         $pipelineRuntime = $pipelineClassReflection->newInstanceWithoutConstructor();
193:         $pipelineReflection = new \ReflectionObject($pipelineRuntime);
194:         
195:         $uniqueIDProperty = $pipelineReflection->getProperty('uniqCode');
196:         $uniqueIDProperty->setAccessible(true);
197:         $uniqueIDProperty->setValue($pipelineRuntime, $uniqueID);
198:         
199:         $statusProperty = $pipelineReflection->getProperty('status');
200:         $statusProperty->setAccessible(true);
201:         $statusProperty->setValue($pipelineRuntime, $toRestore->get('status'));
202:         
203:         $typeProperty = $pipelineReflection->getProperty('type');
204:         $typeProperty->setAccessible(true);
205:         $typeProperty->setValue($pipelineRuntime, $toRestore->get('type'));
206:         
207:         $priorityProperty = $pipelineReflection->getProperty('priority');
208:         $priorityProperty->setAccessible(true);
209:         $priorityProperty->setValue($pipelineRuntime, $toRestore->get('priority'));
210:         
211:         $creationTimeProperty = $pipelineReflection->getProperty('creationTime');
212:         $creationTimeProperty->setAccessible(true);
213:         $creationTimeProperty->setValue($pipelineRuntime, $toRestore->get('creationTime'));
214:         
215:         $completionReportsProperty = $pipelineReflection->getProperty('completionReports');
216:         $completionReportsProperty->setAccessible(true);
217:         $completionReportsProperty->setValue($pipelineRuntime, $toRestore->get('completionReports'));
218:         
219:         $abortMessageProperty = $pipelineReflection->getProperty('abortMessage');
220:         $abortMessageProperty->setAccessible(true);
221:         $abortMessageProperty->setValue($pipelineRuntime, $toRestore->get('abortMessage'));
222:         
223:         $pipelineProperty = $pipelineReflection->getProperty('pipeline');
224:         $pipelineProperty->setAccessible(true);
225:         $pipelineProperty->setValue($pipelineRuntime, PipelineCollector::getPipelineByName($toRestore->get('pipeline')));
226:         
227:         $serializableCollectionProperty = $pipelineReflection->getProperty('serializableCollection');
228:         $serializableCollectionProperty->setAccessible(true);
229:         $serializableCollectionProperty->setValue($pipelineRuntime, new SerializableCollection($toRestore->get('serializableCollection')));
230:         
231:         return $pipelineRuntime;
232:     }
233: }

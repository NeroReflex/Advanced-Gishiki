 21:      * The asymmetric cipher algorithm implementation on a public-key point of view
 22:      *
 23:      * Benato Denis <benato.denis96@gmail.com>
 24:      */
 25:     class AsymmetricPublicKeyCipher
 26:     {
 27: 
 28:         //the public key
 29:         private $publicKey;
 30: 
 31:         /**
 32:          * Create an asymmetric cipher
 33:          */
 34:         public function __construct()
 35:         {
 36:             //initialize an empty public key
 37:             $this->publicKey = null;
 38:         }
 39: 
 40:         /**
 41:          * Check if a valid RSA public key has been loaded
 42:          * 
 43:          * @return bool TRUE if the public key has been loaded, FASE otherwise
 44:          */
 45:         public function IsLoaded()
 46:         {
 47:             return ((gettype($this->publicKey) != "NULL") && (gettype($this->publicKey) == "resource"));
 48:         }
 49: 
 50:         /**
 51:          * Load the public key from a string
 52:          * 
 53:          * @param  string          $keyAsString the key encoded as a string
 54:          * @throws CipherException the exception occurred while importing the key
 55:          */
 56:         public function ImportPublicKey($keyAsString)
 57:         {
 58:             //check if the given string is a loadable key
 59:             if ((strpos($keyAsString, "-----BEGIN PUBLIC KEY-----") !== false) && (strpos($keyAsString, "-----END PUBLIC KEY-----") !== false)) {
 60:                 //if the key seems to be valid load it
 61:                 $this->publicKey = openssl_pkey_get_public($keyAsString);
 62:             } else {
 63:                 throw new CipherException("The RSA exported public key cannot be loaded, because the given data is not a public key", 6);
 64:             }
 65:         }
 66: 
 67:         /**
 68:          * Check if a message was altered using its digital signature (which is provate-key dependant)
 69:          * 
 70:          * @param  string          $message   the messae to be checked
 71:          * @param  string          $signature the digital signature of the message
 72:          * @return bool            verify if the message is correct using its digital signature
 73:          * @throws CipherException the error occurred
 74:          */
 75:         public function VerifyDigitalSignature($message, $signature)
 76:         {
 77:             if (gettype($this->publicKey) != "NULL") {
 78:                 $bynary_non_safe_signature = base64_decode($signature);
 79: 
 80:                 $verificationResult = openssl_verify($message, $bynary_non_safe_signature, $this->publicKey, OPENSSL_ALGO_SHA256);
 81: 
 82:                 //select what must be returned
 83:                 switch ($verificationResult) {
 84:                     case 0:
 85:                         return false;
 86:                     case 1:
 87:                         return true;
 88:                     default:
 89:                         throw new CipherException("An unknown error has occurred while verifying the digital signature", 1);
 90:                 }
 91:             } else {
 92:                 throw new CipherException("The digital signature cannot be verified, because the public key wasn't loaded", 0);
 93:             }
 94:         }
 95: 
 96:         /**
 97:          * Encrypt a message using the loaded public key
 98:          * 
 99:          * @param  string          $message the message to encrypt
100:          * @return string          the encrypted message in a binary safe format 
101:          * @throws CipherException the error occurred
102:          */
103:         public function Encrypt($message)
104:         {
105:             if (gettype($this->publicKey) != "NULL") {
106:                 //the encrypted message
107:                 $encrypted = null;
108: 
109:                 //encrypt the message and check for failure
110:                 if (!openssl_public_encrypt($message, $encrypted, $this->publicKey, OPENSSL_PKCS1_PADDING)) {
111:                     throw new CipherException("OpenSSL couldn't encrypt the given message", 3);
112:                 }
113: 
114:                 //return the ecnrypted message base64-encoded
115:                 return base64_encode($encrypted);
116:             } else {
117:                 throw new CipherException("The plain message cannot be encrypted, because the public key wasn't loaded", 2);
118:             }
119:         }
120: 
121:         /**
122:          * Decrypt a message encrypted with the private key using the loaded public key
123:          * 
124:          * @param  string          $encMessage the encrypted message (private key)
125:          * @return string          the plain message before the encrypted
126:          * @throws CipherException the error occurred
127:          */
128:         public function Decrypt($encMessage)
129:         {
130:             if (gettype($this->publicKey) != "NULL") {
131:                 //the encrypted message
132:                 $plainMessage = null;
133: 
134:                 //encrypt the message and check for failure
135:                 if (!openssl_public_decrypt(base64_decode($encMessage), $plainMessage, $this->publicKey, OPENSSL_PKCS1_PADDING)) {
136:                     throw new CipherException("OpenSSL couldn't decrypt the given message", 5);
137:                 }
138: 
139:                 //return the ecnrypted message
140:                 return $plainMessage;
141:             } else {
142:                 throw new CipherException("The encrypted message cannot be decrypted, because the public key wasn't loaded", 4);
143:             }
144:         }
145: 
146:         /**
147:          * Free the memory used by OpenSSL to hold the public key
148:          */
149:         public function __destruct()
150:         {
151:             //free the public key (if any)
152:             if (gettype($this->publicKey) == "resource") {
153:                 openssl_free_key($this->publicKey);
154:             }
155:         }
156:     }

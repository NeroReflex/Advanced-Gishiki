 22: /**
 23:  * This class represents a private key for the asymmetric encryption engine.
 24:  *
 25:  * Note: This class uses OpenSSL for strong encryption
 26:  *
 27:  * @author Benato Denis <benato.denis96@gmail.com>
 28:  */
 29: final class PrivateKey
 30: {
 31:     //RSA keys length
 32:     const RSA512 = 512;
 33:     const RSA1024 = 1024;
 34:     const RSA2048 = 2048;
 35:     const RSA4096 = 4096;
 36:     const RSAEXTREME = 16384;
 37: 
 38:     /**
 39:      * Create a random private key of the given length (in bits).
 40:      *
 41:      * You can use predefined constants to have valid keylength values.
 42:      *
 43:      * The higher the key length, the higher the security the higher the required time to generate the key.
 44:      *
 45:      * Usage example:
 46:      *
 47:      * <code>
 48:      * //give a name to the file containing the generation result
 49:      * $filename = APPLICATION_DIR."newkey.private.pem";
 50:      *
 51:      * //generate the new key
 52:      * $serailized_key = PrivateKey::Generate(PrivateKey::RSA4096);
 53:      *
 54:      * //export to file the serialized key
 55:      * file_put_contents($filename, $serailized_key);
 56:      * //NOTE: this example is really BAD because the file is NOT encrypted
 57:      *
 58:      * //yes, you can load private keys directly from file
 59:      * $private_key = new PrivateKey("file://".$filename);
 60:      * </code>
 61:      *
 62:      * @param int $keyLength the length (in bits) of the private key to e generated
 63:      *
 64:      * @return string the serialized private key
 65:      *
 66:      * @throws \InvalidArgumentException the given key length is not an integer power of two
 67:      * @throws AsymmetricException       the error occurred while generating and exporting the new private key
 68:      */
 69:     public static function Generate($keyLength = self::RSA4096)
 70:     {
 71:         if (!is_integer($keyLength)) {
 72:             throw new \InvalidArgumentException('The key length must be given as an integer number which is a power of two');
 73:         } elseif ((($keyLength & ($keyLength - 1)) != 0) || ($keyLength == 0)) {
 74:             throw new \InvalidArgumentException('The key length must be a power of two');
 75:         }
 76: 
 77:         //if the key length is extremely long, a very long time will be needed to generate the key, and it is necessary ti set no time limits to generate it
 78:         //if ($keyLength == self::RSAEXTREME) {
 79:             set_time_limit(0);
 80:         //}
 81: 
 82:         //build the configuration array
 83:         $config = [
 84:             'digest_alg' => 'sha512',
 85:             'private_key_bits' => $keyLength,
 86:             'private_key_type' => OPENSSL_KEYTYPE_RSA,
 87:         ];
 88:         //use the application openssl configuration
 89:         if (!is_null(Environment::GetCurrentEnvironment())) {
 90:             $config = array_merge(
 91:                     $config,
 92:                     ['config' => (file_exists(Environment::GetCurrentEnvironment()->GetConfigurationProperty('APPLICATION_DIR').'openssl.cnf')) ?
 93:                         Environment::GetCurrentEnvironment()->GetConfigurationProperty('APPLICATION_DIR').'openssl.cnf' : null, ]);
 94:         }
 95: 
 96:         //create a new private key
 97:         $privateKey = openssl_pkey_new($config);
 98: 
 99:         //check the result
100:         if (!is_resource($privateKey)) {
101:             throw new AsymmetricException('The key generation is not possible due to an unknown '.openssl_error_string(), 8);
102:         }
103: 
104:         //extract the private key string-encoded from the generated private key
105:         $pKeyEncoded = '';
106:         openssl_pkey_export($privateKey, $pKeyEncoded, null, $config);
107: 
108:         //free the memory space used to hold the newly generated key
109:         openssl_free_key($privateKey);
110: 
111:         //check for the result
112:         if (!is_string($pKeyEncoded)) {
113:             throw new AsymmetricException("The key generation was completed, but the result couldn't be exported", 9);
114:         }
115: 
116:         //return the operation result
117:         return $pKeyEncoded;
118:     }

127:      * @var resource the private key ready to be used by OpenSSL
128:      */

131:     /**
132:      * Used to create a private key from the given string.
133:      *
134:      * If a string containing a serialized private key is
135:      * not give, the framework default one will be used
136:      *
137:      * @param string|null $customKey         the private key serialized as a string
138:      * @param string      $customKeyPassword the password to decrypt the serialized private key (if necessary)
139:      *
140:      * @throws \InvalidArgumentException the given key and/or password isn't a valid string
141:      * @throws AsymmetricException       the given key is invalid
142:      */
143:     public function __construct($customKey = null, $customKeyPassword = '')
144:     {
145:         if (!is_string($customKeyPassword)) {
146:             throw new \InvalidArgumentException('The private key password cannot be something else than a string');
147:         }
148: 
149:         if ((!is_string($customKey)) && (!is_null($customKey))) {
150:             throw new \InvalidArgumentException('The serialized private key must be a string');
151:         }
152: 
153:         //get a string containing a serialized asymmetric key
154:         $serialized_key = (is_string($customKey)) ?
155:             $serialized_key = $customKey :
156:             Environment::GetCurrentEnvironment()->GetConfigurationProperty('MASTER_ASYMMETRIC_KEY');
157: 
158:         //get the beginning and ending of a private key (to stip out additional shit and check for key validity)
159:         $is_encrypted = strpos($serialized_key, 'ENCRYPTED') !== false;
160: 
161:         //get the password of the serialized key
162:         $serializedKeyPassword = ($is_encrypted) ? $customKeyPassword : '';
163: 
164:         //load the private key
165:         $this->key = openssl_pkey_get_private($serialized_key, $serializedKeyPassword);
166: 
167:         //check for errors
168:         if (!$this->isLoaded()) {
169:             throw new AsymmetricException('The private key could not be loaded', 0);
170:         }
171:     }

174:      * Free resources used to hold this private key.
175:      */
176:     public function __destruct()
177:     {
178:         if ($this->isLoaded()) {
179:             openssl_free_key($this->key);
180:         }

184:      * Export the public key corresponding to this private key.
185:      *
186:      * @return string the public key exported from this private key
187:      */
188:     public function exportPublicKey()
189:     {
190:         //get details of the current private key
191:         $privateKeyDetails = openssl_pkey_get_details($this->key);
192:

194:         return $privateKeyDetails['key'];

197:     /**
198:      * Export this private key in a string format.
199:      *
200:      * The resulting string can be used to construct another PrivateKey instance:
201:      *
202:      * <code>
203:      * use Gishiki\Security\Encryption\Asymmetric\PrivateKey;
204:      *
205:      * //this is the exported private key
206:      * $exported_key = "...";
207:      *
208:      * //rebuild the private key
209:      * $privateKey = new PrivateKey($exported_key);
210:      * </code>
211:      *
212:      * @param string $keyPassword the private key password
213:      *
214:      * @return string the serialized private key
215:      *
216:      * @throws \InvalidArgumentException the given password is not a string
217:      * @throws AsymmetricException       the given key is invalid
218:      */
219:     public function export($keyPassword = '')
220:     {
221:         if (!is_string($keyPassword)) {
222:             throw new \InvalidArgumentException('The private key password cannot be something else than a string');
223:         } elseif (!$this->isLoaded()) {
224:             throw new AsymmetricException('It is impossible to serialize an unloaded private key: '.openssl_error_string(), 1);
225:         }
226: 
227:         $serialized_key = '';
228: 
229:         //build the configuration array
230:         $config = [
231:             'digest_alg' => 'sha512',
232:             'private_key_type' => OPENSSL_KEYTYPE_RSA,
233:         ];
234:         //use the application openssl configuration
235:         if (!is_null(Environment::GetCurrentEnvironment())) {
236:             $config = array_merge(
237:                     $config,
238:                     ['config' => (file_exists(Environment::GetCurrentEnvironment()->GetConfigurationProperty('APPLICATION_DIR').'openssl.cnf')) ?
239:                         Environment::GetCurrentEnvironment()->GetConfigurationProperty('APPLICATION_DIR').'openssl.cnf' : null, ]);
240:         }
241: 
242:         //serialize the key and encrypt it if requested
243:         if (strlen($keyPassword) > 0) {
244:             openssl_pkey_export($this->key, $serialized_key, $keyPassword, $config);
245:         } else {
246:             openssl_pkey_export($this->key, $serialized_key, null, $config);
247:         }
248: 
249:         //return the serialized key
250:         return $serialized_key;
251:     }

254:      * Check if the key has been loaded.
255:      *
256:      * @return bool true if the key has been loaded
257:      */
258:     public function isLoaded()
259:     {
260:         return is_resource($this->key);
261:     }

276:      * @return array the array that contains the key and its legth (in bytes)
277:      *
278:      * @throws AsymmetricException the key cannot be exported
279:      */
280:     public function __invoke()
281:     {
282:         if (!$this->isLoaded()) {
283:             throw new AsymmetricException('It is impossible to obtain an unloaded private key', 1);
284:         }
285: 
286:         //get private key details
287:         $details = openssl_pkey_get_details($this->key);
288: 
289:         return [
290:             'key' => &$this->key,
291:             'byteLength' => $details['bits'] / 8,
292:         ];
293:     }

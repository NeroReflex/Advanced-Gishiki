 20: /**
 21:  * This class is a collection of supported algorithms.
 22:  * 
 23:  * Note: This class uses OpenSSL for strong encryption
 24:  *
 25:  * @author Benato Denis <benato.denis96@gmail.com>
 26:  */
 27: abstract class Algorithms
 28: {
 29:     /**************************************************************************
 30:      *                     Common hashing algorithms                          *
 31:      **************************************************************************/
 32:     const CRC32 = 'crc32';
 33:     const MD4 = 'md4';
 34:     const MD5 = 'md5';
 35:     const SHA1 = 'sha1';
 36:     const SHA256 = 'sha256';
 37:     const SHA328 = 'sha384';
 38:     const SHA512 = 'sha512';
 39:     const ROT13 = 'rot13';
 40: 
 41:     /**
 42:      * Generate the message digest for the given message.
 43:      * 
 44:      * An example usage is:
 45:      * 
 46:      * <code>
 47:      * $message = "this is the message to be hashed";
 48:      * 
 49:      * $test_gishiki_md5 = Algorithms::hash($message, Algorithms::MD5);
 50:      * $test_php_md5 = md5($message);
 51:      * 
 52:      * if ($test_gishiki_md5 == $test_php_md5) {
 53:      *     echo "Gishiki's MD5 produces the same exact hash of the PHP's MD5";
 54:      * }
 55:      * </code>
 56:      * 
 57:      * @param string $message      the string to be hashed
 58:      * @param string $algorithm    the name of the hashing algorithm
 59:      * @param bool   $binaryUnsafe if false the result is binhex
 60:      *
 61:      * @return string the result of the hash algorithm
 62:      *
 63:      * @throws \InvalidArgumentException the message or the algorithm is given as a non-string or an empty string
 64:      * @throws HashingException          the error occurred while generating the hash for the given message
 65:      */
 66:     public static function hash($message, $algorithm = self::MD5, $binaryUnsafe = false)
 67:     {
 68:         //check for the parameter
 69:         if (!is_bool($binaryUnsafe)) {
 70:             throw new \InvalidArgumentException('The binary safeness must be given as a boolean value');
 71:         }
 72: 
 73:         //check for the message
 74:         if ((!is_string($message)) || (strlen($message) <= 0)) {
 75:             throw new \InvalidArgumentException('The message to be hashed must be given as a valid non-empty string');
 76:         }
 77: 
 78:         //check for the algorithm name
 79:         if ((!is_string($algorithm)) || (strlen($algorithm) <= 0)) {
 80:             throw new \InvalidArgumentException('The name of the hashing algorithm must be given as a valid non-empty string');
 81:         }
 82: 
 83:         if ($algorithm == self::ROT13) {
 84:             return str_rot13($message);
 85:         }
 86: 
 87:         //check if the hashing algorithm is supported
 88:         if ((!in_array($algorithm, openssl_get_md_methods())) && (!in_array($algorithm, hash_algos()))) {
 89:             throw new HashingException('An error occurred while generating the hash, because an unsupported hashing algorithm has been selected', 0);
 90:         }
 91: 
 92:         //calculate the hash for the given message
 93:         $hash = ((in_array($algorithm, openssl_get_md_methods()))) ? openssl_digest($message, $algorithm, $binaryUnsafe) : hash($algorithm, $algorithm, $binaryUnsafe);
 94: 
 95:         //check for errors
 96:         if ($hash === false) {
 97:             throw new HashingException('An unknown error occurred while generating the hash', 1);
 98:         }
 99: 
100:         //return the calculated message digest
101:         return $hash;
102:     }
103: 
104:     /**
105:      * PBKDF2 key derivation function as defined by RSA's PKCS #5: https://www.ietf.org/rfc/rfc2898.txt.
106:      * 
107:      * Test vectors can be found here: https://www.ietf.org/rfc/rfc6070.txt
108:      *
109:      * This implementation of PBKDF2 was originally created by https://defuse.ca
110:      * With improvements by http://www.variations-of-shadow.com
111:      * 
112:      * @param string $password     the password
113:      * @param string $salt         a salt that is unique to the password
114:      * @param string $keyLength    the length of the derived key in bytes
115:      * @param string $count        iteration count. Higher is better, but slower. Recommended: At least 1000
116:      * @param string $algorithm    the hash algorithm to use. Recommended: SHA256
117:      * @param string $binaryUnsafe if true, the key is returned in raw binary format. Hex encoded otherwise
118:      *
119:      * @return string the key derived from the password and salt
120:      *
121:      * @throws \InvalidArgumentException invalid arguments have been passed
122:      * @throws HashingException          the error occurred while generating the requested hashing algorithm
123:      */
124:     public static function pbkdf2($password, $salt, $keyLength, $count, $algorithm = self::SHA1, $binaryUnsafe = false)
125:     {
126:         if ((!is_integer($count)) || ($count <= 0)) {
127:             throw new \InvalidArgumentException('The iteration number for the PBKDF2 function must be a positive non-zero integer', 2);
128:         }
129: 
130:         if ((!is_integer($keyLength)) || ($keyLength <= 0)) {
131:             throw new \InvalidArgumentException('The resulting key length for the PBKDF2 function must be a positive non-zero integer', 2);
132:         }
133: 
134:         if ((!is_string($algorithm)) || (strlen($algorithm) <= 0)) {
135:             throw new \InvalidArgumentException('The hashing algorithm for the PBKDF2 function must be a non-empty string', 2);
136:         }
137: 
138:         //an algorithm is represented as a string of only lowercase chars
139:         $algorithm = strtolower($algorithm);
140: 
141:         //the raw output of the max legth (beyond the $keyLength algorithm)
142:         $output = '';
143: 
144:         if (function_exists('openssl_pbkdf2')) {
145:             /*          execute the native openssl_pbkdf2                   */
146: 
147:             //check if the algorithm is valid
148:             if (!in_array($algorithm, openssl_get_md_methods(true), true)) {
149:                 throw new HashingException('Invalid algorithm: the choosen algorithm is not valid for the PBKDF2 function', 2);
150:             }
151: 
152:             $output = openssl_pbkdf2($password, $salt, $keyLength, $count, $algorithm);
153:         } elseif (function_exists('hash_pbkdf2')) {
154:             /*          execute the native hash_pbkdf2                     */
155: 
156:             //check if the algorithm is valid
157:             if (!in_array($algorithm, hash_algos(), true)) {
158:                 throw new HashingException('Invalid algorithm: the choosen algorithm is not valid for the PBKDF2 function', 2);
159:             }
160: 
161:             // The output length is in NIBBLES (4-bits) if $binaryUnsafe is false!
162:             if (!$binaryUnsafe) {
163:                 $keyLength = $keyLength * 2;
164:             }
165: 
166:             return hash_pbkdf2($algorithm, $password, $salt, $count, $keyLength, $binaryUnsafe);
167:         } else {
168:             /*          use an hack to emulate openssl_pbkdf2               */
169: 
170:             //check if the algorithm is valid
171:             if (!in_array($algorithm, hash_algos(), true)) {
172:                 throw new HashingException('Invalid algorithm: the choosen algorithm is not valid for the PBKDF2 function', 2);
173:             }
174: 
175:             $hashLength = strlen(hash($algorithm, '', true));
176:             $blockCount = ceil($keyLength / $hashLength);
177: 
178:             for ($i = 1; $i <= $blockCount; ++$i) {
179:                 // $i encoded as 4 bytes, big endian.
180:                 $last = $salt.pack('N', $i);
181: 
182:                 // first iteration
183:                 $last = $xorsum = hash_hmac($algorithm, $last, $password, true);
184: 
185:                 // perform the other $count - 1 iterations
186:                 for ($j = 1; $j < $count; ++$j) {
187:                     $xorsum ^= ($last = hash_hmac($algorithm, $last, $password, true));
188:                 }
189:                 $output .= $xorsum;
190:             }
191:         }
192: 
193:         return ($binaryUnsafe) ? substr($output, 0, $keyLength) : bin2hex(substr($output, 0, $keyLength));
194:     }
195: }

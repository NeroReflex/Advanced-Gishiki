 18: namespace Gishiki\Security {
 19:     
 20:     /**
 21:      * An helper class for symmetric cipher algorithms
 22:      *
 23:      * @author Benato Denis <benato.denis96@gmail.com>
 24:      */
 25:     class SymmetricCipher
 26:     {
 27:         //crypting algorythms
 28:         const AES128 = 16;
 29:         const AES192 = 24;
 30:         const AES256 = 32;
 31: 
 32:         /** the algorithm to be used */
 33:         private $algorithm;
 34: 
 35:         /** the key that will be used */
 36:         private $key; /*note that this can be used even as key length*/
 37: 
 38:         /** the openssl internal algorithm name */
 39:         private $algorithmName; /*used to speedup the source*/
 40: 
 41:         /**
 42:          * Create a ciper that uses the given algorithm/key length
 43:          * 
 44:          * @param  int             $encAlg the key length for AES (default: ::AES128)
 45:          * @throws CipherException the error occurred
 46:          */
 47:         public function __construct($encAlg = SymmetricCipher::AES128)
 48:         {
 49:             //store the encryption algorith
 50:             $this->algorithm = $encAlg;
 51: 
 52:             //set the default password
 53:             $this->SetKey(Environment::GetCurrentEnvironment()->GetConfigurationProperty('SECURITY_MASTER_SYMMETRIC_KEY'));
 54: 
 55:             //get the encryption name
 56:             switch ($this->algorithm) {
 57:                 case self::AES128:
 58:                     $this->algorithmName = 'aes-128-cbc';
 59:                     break;
 60:                 case self::AES192:
 61:                     $this->algorithmName = 'aes-192-cbc';
 62:                     break;
 63:                 case self::AES256:
 64:                     $this->algorithmName = 'aes-256-cbc';
 65:                     break;
 66:                 default:
 67:                     throw new CipherException("The given algoryth/key length is not valid", 8);
 68:             }
 69:         }
 70: 
 71:         /**
 72:          * Set a new key for encryptions and decryptions
 73:          * @param string $key the new key
 74:          */
 75:         public function SetKey($key)
 76:         {
 77:             //generate and store the key
 78:             $this->key = substr($key.sha1($key).md5($key), 0, $this->algorithm);
 79: 
 80:             /*openssl_pbkdf2($key, mb_substr(strtohex($Salt), 0, 8, 'UTF-8'), $this->algorithm, 1000, $keySize); is not supported on all PHP framework target versions*/
 81:         }
 82: 
 83:         /**
 84:          * Encrypt a message
 85:          * 
 86:          * @param  string $plainData the original string/data to be encrypted
 87:          * @return string the encrypted data
 88:          */
 89:         public function Encrypt($plainData)
 90:         {
 91:             //generate and store a random IV
 92:             $iv = openssl_random_pseudo_bytes(openssl_cipher_iv_length($this->algorithmName));
 93: 
 94:             //get the encrypted data
 95:             $encrypted = openssl_encrypt($plainData, $this->algorithmName, $this->key, 0, $iv);
 96: 
 97:             //encode encrypted data and the random IV with Base64 and add it to the encryption result
 98:             $encrypted = base64_encode($iv).":>=|=<:".base64_encode($encrypted);
 99: 
100:             //return the encoded string
101:             return $encrypted;
102:         }
103: 
104:         /**
105:          * Decrypt a message encrypted by the Encrypt function
106:          * @param  string          $cipedData the result of the encryption
107:          * @return string          the original data
108:          * @throws CipherException the error occurred
109:          */
110:         public function Decrypt($cipedData)
111:         {
112:             //get IV and encrypted text in an array
113:             $dataToEncrypt = explode(":>=|=<:", $cipedData, 2);
114: 
115:             //check for the data to be valid
116:             if (count($dataToEncrypt) != 2) {
117:                 throw new CipherException("The content is not a valid encrypted content", 9);
118:             }
119: 
120:             //rebuild the random IV used to encrypt the string
121:             $iv = base64_decode($dataToEncrypt[0]);
122: 
123:             //get the data to decrypt
124:             $encrypted = base64_decode($dataToEncrypt[1]);
125: 
126:             //decrypt and return the result
127:             return openssl_decrypt($encrypted, $this->algorithmName, $this->key, 0, $iv);
128:         }
129:     }
130: }

18: namespace Gishiki\Logging\Adapter;
19: 
20: /**
21:  * An helper class for storing logs of what happens on the server.
22:  * 
23:  * This is the gelf version of the logger (based on gelf-php project)
24:  *
25:  * @link https://github.com/bzikarsky/gelf-php
26:  * 
27:  * Benato Denis <benato.denis96@gmail.com>
28:  */
29: class GelfAdapter
30: {
31:     //this is the managed gelf resource
32:     private $gelf_resource = null;
33: 
34:     /**
35:      * Setup a logger that works on gelf.
36:      * 
37:      * This requires a graylog server to be waiting for logs.
38:      * 
39:      * Default server is on udp:localhost:12201
40:      * 
41:      * @param string $server the address of the server
42:      */
43:     public function __construct($server = 'null')
44:     {
45:         $native_server = ($server == 'null') ? null : $server;
46: 
47:         if ($native_server) {
48:             $conn_info = explode(':', $native_server);
49: 
50:             $transport = null;
51:             switch ($conn_info[0]) {
52:                 case 'tcp' :
53:                     $transport = new \Gelf\Transport\TcpTransport($conn_info[1], intval($conn_info[2]));
54:                     break;
55: 
56:                 case 'udp':
57:                     $transport = new \Gelf\Transport\UdpTransport($conn_info[1], intval($conn_info[2]));
58:                     break;
59: 
60:                 default:
61:                     $transport = null;
62:             }
63: 
64:             //create the new logger with the build transport
65:             $this->gelf_resource = new \Gelf\Logger($transport);
66:         } else {
67:             $this->gelf_resource = new \Gelf\Logger();
68:         }
69:     }
70: 
71:     public function log($level, $message, array $context = array())
72:     {
73:         //return value isn't documentated because it MUST NOT be used/trusted
74:         return ($this->gelf_resource) ? $this->gelf_resource->log($level, $message, $context) : null;
75:     }
76: }

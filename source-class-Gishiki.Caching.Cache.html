 20:     /**
 21:      * Provide basic caching using memcached or redis.
 22:      *
 23:      * @author Benato Denis <benato.denis96@gmail.com>
 24:      */
 25:     abstract class Cache
 26:     {
 27:         private static $persistence_ID = "lf4e2d";
 28:         private static $connected = FALSE;
 29: 
 30:         /**
 31:          * The connection to the cache server specified in the application configuration file.
 32:          */
 33:         protected static $cacheServer = [];
 34: 
 35:         /**
 36:          * Check for the availability of a cache server/container
 37:          * 
 38:          * @return boolean TRUE if a cache container is actually connected
 39:          */
 40:         public static function Connected() {
 41:             return (self::$connected);
 42:         }
 43:         
 44:         /**
 45:          * Initialize the caching engine for the current request.
 46:          * This function is automatically called by the framework.
 47:          * Another call to this function won't produce any effects.
 48:          */
 49:         public static function Initialize() {
 50:             if (!self::$connected) {
 51:                 //initialize the caching engine only if it is needed
 52:                 if (\Gishiki\Core\Environment::GetCurrentEnvironment()->GetConfigurationProperty('CACHING_ENABLED')) {
 53:                     //parse the string stored to the application settings file
 54:                     self::$cacheServer["details"] = CacheConnectionString::Parse(\Gishiki\Core\Environment::GetCurrentEnvironment()->GetConfigurationProperty("CACHE_CONNECTION_STRING"));
 55: 
 56:                     switch (self::$cacheServer["details"]["server_type"]) {
 57:                         case "memcached":
 58:                             if (\Gishiki\Core\Environment::ExtensionSupport("MEMCACHED")) {
 59:                                 //open a new memcached instance
 60:                                 self::$cacheServer["connection"] = new \Memcached(self::$persistence_ID);
 61:                                 self::$cacheServer["connection"]->setOption(\Memcached::OPT_RECV_TIMEOUT, 1000);
 62:                                 self::$cacheServer["connection"]->setOption(\Memcached::OPT_SEND_TIMEOUT, 1000);
 63:                                 self::$cacheServer["connection"]->setOption(\Memcached::OPT_TCP_NODELAY, true);
 64:                                 self::$cacheServer["connection"]->setOption(\Memcached::OPT_SERVER_FAILURE_LIMIT, 50);
 65:                                 self::$cacheServer["connection"]->setOption(\Memcached::OPT_CONNECT_TIMEOUT, 500);
 66:                                 self::$cacheServer["connection"]->setOption(\Memcached::OPT_RETRY_TIMEOUT, 300);
 67: 
 68:                                 //suppress a warning on Memcached::append()
 69:                                 self::$cacheServer["connection"]->setOption(\Memcached::OPT_COMPRESSION, false);
 70: 
 71:                                 //connect to the memcached server
 72:                                 self::$cacheServer["connection"]->addServer(self::$cacheServer["details"]["server_address"], self::$cacheServer["details"]["server_port"]);
 73: 
 74:                                 //was the server connected?
 75:                                 self::$connected = count(self::$cacheServer["connection"]->getStats()) > 0;
 76:                             }
 77:                             break;
 78: 
 79:                         case "filesystem":
 80:                             self::$connected = (file_exists(self::$cacheServer["details"]["directory"])) && (is_writable(self::$cacheServer["details"]["directory"]));
 81:                             break;
 82: 
 83:                         default:
 84:                             break;
 85:                     }
 86:                 }
 87:             }
 88:         }
 89: 
 90:         /**
 91:          * Store data on the cache server. Complex elaboration results should be stored into the cache,
 92:          * This way a lot of computational time can be saved!
 93:          *
 94:          * @param string $cacheName the name of the cache fragment
 95:          * @param mixed $cacheValue the value of the cache fragment
 96:          */
 97:         static function Store($cacheName, $cacheValue) {
 98:             //if a caching server is connected, and the cache fragment has a valid name
 99:             if ((self::$connected) && (gettype($cacheName) == "string") && ($cacheName != "")) {
100: 
101:                 //chose the proper way of storing the cache fragment
102:                 switch (self::$cacheServer["details"]["server_type"]) {
103:                     case "memcached":
104:                         self::$cacheServer["connection"]->set($cacheName, serialize($cacheValue));
105:                         break;
106: 
107:                     case "filesystem":
108:                         $filename = self::$cacheServer["details"]["directory"].self::$persistence_ID.md5($cacheName).".cachefragment";
109:                         file_put_contents($filename, $cacheValue, LOCK_EX);
110:                         break;
111:                 }
112:             }
113:         }
114: 
115:         /**
116:          * Check if a cache fragment with the given name exists in the cache server
117:          *
118:          * @param string $cacheName the name of the cache fragment
119:          * @return boolean TRUE if the cache fragment exists, FALSE otherwise
120:          */
121:         public static function Exists($cacheName) {
122:             //is the cache fragment existent?
123:             $exists = FALSE;
124: 
125:             //if a caching server is connected, and the cache fragment has a valid name
126:             if ((self::$connected) && (gettype($cacheName) == "string") && ($cacheName != "")) {
127: 
128:                 //chose the proper way of checking the cache fragment
129:                 switch (self::$cacheServer["details"]["server_type"]) {
130:                     case "memcached":
131:                         if (self::$cacheServer["connection"]->append($cacheName, "") != TRUE)
132:                             return self::$cacheServer["connection"]->getResultCode() !== \Memcached::RES_NOTSTORED;
133:                         else
134:                             return TRUE;
135:                         break;
136: 
137:                     case "filesystem":
138:                         return file_exists(self::$cacheServer["details"]["directory"].self::$persistence_ID.md5($cacheName).".cachefragment");
139:                 }
140:             }
141: 
142:             return $exists;
143:         }
144: 
145:         /**
146:          * Fetch the value of the cache fragment with the given name
147:          *
148:          * @param string $cacheName the name of the cache fragment
149:          * @return mixed the value of the cache fragment can be (string, boolean, integer and float), NULL on error
150:          */
151:         public static function Fetch($cacheName) {
152:             //if a caching server is connected, and the cache fragment has a valid name
153:             if ((self::$connected) && (gettype($cacheName) == "string") && ($cacheName != "")) {
154: 
155:                 //chose the proper way of fetching the cache fragment
156:                 switch (self::$cacheServer["details"]["server_type"]) {
157:                     case "memcached":
158:                         return unserialize(self::$cacheServer["connection"]->get($cacheName));
159: 
160:                     case "filesystem":
161:                         return unserialize(file_get_contents(self::$cacheServer["details"]["directory"].self::$persistence_ID.md5($cacheName).".cachefragment", FALSE));
162:                 }
163:             }
164: 
165:             return NULL;
166:         }
167: 
168:         /**
169:          * Delete the cache fragment with the given name
170:          *
171:          * @param string $cacheName the name of the cache fragment
172:          */
173:         public static function Delete($cacheName) {
174:             //if a caching server is connected, and the cache fragment has a valid name
175:             if ((self::$connected) && (gettype($cacheName) == "string") && ($cacheName != "")) {
176: 
177:                 //chose the proper way of removing the cache fragment
178:                 switch (self::$cacheServer["details"]["server_type"]) {
179:                     case "memcached":
180:                         self::$cacheServer["connection"]->delete($cacheName);
181:                         break;
182: 
183:                     case "filesystem":
184:                         if (self::Exists($cacheName))
185:                         {   unlink(self::$cacheServer["details"]["directory"].self::$persistence_ID.md5($cacheName).".cachefragment");  }
186:                         break;
187:                 }
188:             }
189:         }
190:         
191:         /**
192:          * Invalidate all items in the cache
193:          */
194:         public static function Flush() {
195:             //if a caching server is connected, and the cache fragment has a valid name
196:             if ((self::$connected) && (gettype($cacheName) == "string") && ($cacheName != "")) {
197: 
198:                 //chose the proper way of removing the cache fragment
199:                 switch (self::$cacheServer["details"]["server_type"]) {
200:                     case "memcached":
201:                         self::$cacheServer["connection"]->flush();
202:                         break;
203: 
204:                     case "filesystem":
205:                         if (self::Exists($cacheName)) {
206:                             unlink(self::$cacheServer["details"]["directory"]);
207:                             mkdir(self::$cacheServer["details"]["directory"]);
208:                         }
209:                         break;
210:                 }
211:             }
212:         }

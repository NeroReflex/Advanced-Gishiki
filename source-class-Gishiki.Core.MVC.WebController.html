 18: namespace Gishiki\Core\MVC {
 19:     
 20:     /**
 21:      * The Gishiki base web controller. Every web controller (controllers used to 
 22:      * generate an application for the prowser) inherit from this class
 23:      * 
 24:      * @author Benato Denis <benato.denis96@gmail.com>
 25:      */
 26:     class WebController {
 27:         /** this is the HTML that will be sent to the client */
 28:         private $rawContent;
 29:         
 30:         /**
 31:          * Initialize the web controller. Each web controller MUST call this constructor
 32:          */
 33:         public function __construct() {
 34:             //load an empty response buffer
 35:             $this->rawContent = "";
 36:         }
 37:         
 38:         /**
 39:          * Load a template inside the HTML response buffer
 40:          * 
 41:          * @param string $templateName the name of the page template
 42:          * @throws \Exception an exception is thrown if the template cannot be found
 43:          */
 44:         protected function LoadTemplate($templateName) {
 45:             //check for the partial view existence
 46:             if (file_exists(\Gishiki\Core\Environment::GetCurrentEnvironment()->GetConfigurationProperty('VIEW_DIR').$templateName.".template")) {
 47:                 
 48:                 //get the raw partial view
 49:                 $content = "";
 50:                 if (file_exists(\Gishiki\Core\Environment::GetCurrentEnvironment()->GetConfigurationProperty('VIEW_DIR').$templateName.".template")) {
 51:                     $content = file_get_contents(\Gishiki\Core\Environment::GetCurrentEnvironment()->GetConfigurationProperty('VIEW_DIR').$templateName.".template");
 52:                 }
 53: 
 54:                 //for each data subset query update the partial view
 55:                 if (gettype($dataSubset) == "array")
 56:                 {
 57:                     //perform swaps/replacements
 58:                     $substitutions = count($dataSubset);
 59:                     for ($i = 0; $i < $substitutions; $i++) {
 60:                         $currentData = current($dataSubset);
 61:                         $currentDataIndex = key($dataSubset);
 62:                         $content = str_replace("{{".$currentDataIndex."}}", $currentData, $content);
 63:                         next($dataSubset);
 64:                     }
 65:                 }
 66: 
 67:                 //include the template
 68:                 $this->rawContent = $content;
 69:             } else {
 70:                 throw new \Exception("The template '".$templateName."' cannot be found");
 71:             }
 72:         }
 73:         
 74:         /**
 75:          * Process a partial view and store the result to the output buffer 
 76:          * (that will be given to the client at the end of the controller lifetime)
 77:          * 
 78:          * @param string $viewName the name of the partial view WITHOUT '.html'
 79:          * @param array $dataSubset an array of sobstitution strings
 80:          * @param string $viewPlaceHolder this is used to complete the template previously loaded
 81:          * @throws \Exception an exception is thrown if the partial view cannot be found
 82:          */
 83:         protected function LoadView($viewName, $dataSubset = NULL, $viewPlaceHolder = "") {
 84:             //check for the partial view existence
 85:             if (file_exists(\Gishiki\Core\Environment::GetCurrentEnvironment()->GetConfigurationProperty('VIEW_DIR').$viewName.".html")) {
 86:                 
 87:                 //get the raw partial view
 88:                 $content = "";
 89:                 if (file_exists(\Gishiki\Core\Environment::GetCurrentEnvironment()->GetConfigurationProperty('VIEW_DIR').$viewName.".html")) {
 90:                     $content = file_get_contents(\Gishiki\Core\Environment::GetCurrentEnvironment()->GetConfigurationProperty('VIEW_DIR').$viewName.".html");
 91:                 }
 92:                 
 93:                 //perform swaps/replacements
 94:                 foreach ($dataSubset as $currentDataIndex => $currentData) {
 95:                     $content = str_replace("{{".$currentDataIndex."}}", htmlentities($currentData, ENT_HTML5), $content);
 96:                 }
 97: 
 98:                 if (strlen($viewPlaceHolder) > 0) //complete the template if a valid placeholder is given
 99:                 {   $this->rawContent = str_replace("{{{".$viewPlaceHolder."}}}", $content."{{{".$viewPlaceHolder."}}}", $this->rawContent);    }
100:             } else {
101:                 throw new \Exception("The partial view \'".$viewName."\' cannot be found");
102:             }
103:         }
104:         
105:         /**
106:          * Send the result of the controller execution to the browser
107:          */
108:         public function __destruct() {
109:             //call the controller standard destructor
110:             parent::__destruct();
111: 
112:             //delete every content placeholder
113:             $matches = [];
114:             preg_match('/{{{(.*)\?}}}/', $this->rawContent, $matches);
115:             while (count($matches) > 0) //remove any placeholder from the page content and from the array
116:             {   str_replace(array_pop($matches), "", $this->rawContent);    }
117:         }
118:         
119:     }
120: }

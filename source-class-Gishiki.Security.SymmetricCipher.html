 18: namespace Gishiki\Security {

 20:     /**
 21:      * An helper class for symmetric cipher algorithms
 22:      *
 23:      * @author Benato Denis <benato.denis96@gmail.com>
 24:      */
 25:     class SymmetricCipher {
 26:         //crypting algorythms
 27:         const AES128 = 16;
 28:         const AES192 = 24;
 29:         const AES256 = 32;
 30: 
 31:         /** the algorithm to be used */
 32:         private $algorithm;
 33: 
 34:         /** the key that will be used */
 35:         private $key; /*note that this can be used even as key length*/
 36: 
 37:         /** the openssl internal algorithm name */
 38:         private $algorithmName; /*used to speedup the source*/
 39: 
 40:         /**
 41:          * Create a ciper that uses the given algorithm/key length
 42:          * 
 43:          * @param int $encAlg the key length for AES (default: ::AES128)
 44:          * @throws CipherException the error occurred
 45:          */
 46:         public function __construct($encAlg = SymmetricCipher::AES128) {
 47:             //store the encryption algorith
 48:             $this->algorithm = $encAlg;
 49: 
 50:             //set the default password
 51:             $this->SetKey(Environment::GetCurrentEnvironment()->GetConfigurationProperty('SECURITY_MASTER_SYMMETRIC_KEY'));
 52: 
 53:             //get the encryption name
 54:             switch ($this->algorithm) {
 55:                 case self::AES128:
 56:                     $this->algorithmName = 'aes-128-cbc';
 57:                     break;
 58:                 case self::AES192:
 59:                     $this->algorithmName = 'aes-192-cbc';
 60:                     break;
 61:                 case self::AES256:
 62:                     $this->algorithmName = 'aes-256-cbc';
 63:                     break;
 64:                 default:
 65:                     throw new CipherException("The given algoryth/key length is not valid", 8);
 66:             }
 67:         }
 68: 
 69:         /**
 70:          * Set a new key for encryptions and decryptions
 71:          * @param string $key the new key
 72:          */
 73:         public function SetKey($key) {
 74:             //generate and store the key
 75:             $this->key = substr($key.sha1($key).md5($key), 0, $this->algorithm);
 76: 
 77:             /*openssl_pbkdf2($key, mb_substr(strtohex($Salt), 0, 8, 'UTF-8'), $this->algorithm, 1000, $keySize); is not supported on all PHP framework target versions*/
 78:         }
 79: 
 80:         /**
 81:          * Encrypt a message
 82:          * 
 83:          * @param string $plainData the original string/data to be encrypted
 84:          * @return string the encrypted data
 85:          */
 86:         public function Encrypt($plainData) {
 87:             //generate and store a random IV
 88:             $iv = openssl_random_pseudo_bytes(openssl_cipher_iv_length($this->algorithmName));
 89: 
 90:             //get the encrypted data
 91:             $encrypted = openssl_encrypt($plainData, $this->algorithmName, $this->key, 0, $iv);
 92: 
 93:             //encode encrypted data and the random IV with Base64 and add it to the encryption result
 94:             $encrypted = base64_encode($iv).":>=|=<:".base64_encode($encrypted);
 95: 
 96:             //return the encoded string
 97:             return $encrypted;
 98:         }
 99: 
100:         /**
101:          * Decrypt a message encrypted by the Encrypt function
102:          * @param string $cipedData the result of the encryption
103:          * @return string the original data
104:          * @throws CipherException the error occurred
105:          */
106:         public function Decrypt($cipedData) {
107:             //get IV and encrypted text in an array
108:             $dataToEncrypt = explode(":>=|=<:", $cipedData, 2);
109: 
110:             //check for the data to be valid
111:             if (count($dataToEncrypt) != 2) {
112:                 throw new CipherException("The content is not a valid encrypted content", 9);
113:             }
114: 
115:             //rebuild the random IV used to encrypt the string
116:             $iv = base64_decode($dataToEncrypt[0]);
117: 
118:             //get the data to decrypt
119:             $encrypted = base64_decode($dataToEncrypt[1]);
120: 
121:             //decrypt and return the result
122:             return openssl_decrypt($encrypted, $this->algorithmName, $this->key, 0, $iv);
123:         }
124:     }
